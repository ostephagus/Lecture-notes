Consider a setup as in figure~\ref{figWaterWaves}. We let $\phi$ be the velocity potential, and $z = \zeta(x, y, t)$ be the upper free surface of water of depth $h$. We define $z = 0$ to be the equilibrium position of the free surface.
\subsection{Governing equations}
We will assume that the flow is inviscid and irrotational for all time. Therefore $\vec{u} = \nabla\phi$ and on the given domain,
\begin{equation}
    \nabla^2 \phi = 0
    \label{eqnLaplace}
\end{equation}
Applying the Kinematic Boundary Condition at the bottom, $\vec{u} \cdot \vec{n} = \vec{u} \cdot \nabla \phi = 0$, that is,
\begin{equation}
    \left.\frac{\partial \phi}{\partial z}\right|_{z = -h} = 0
    \label{eqnBottomKBC}
\end{equation}
The boundary condition for $\phi$ on the free surface was derived in \hyperref[secKBC]{section~\ref{secKBC}}:
\begin{equation}
    \frac{\partial \zeta}{\partial t} + u_x \frac{\partial \zeta}{\partial x} + u_y \frac{\partial \zeta}{\partial y} = w \text{ at } z = \zeta
    \label{eqnFreeSurfaceKBC}
\end{equation}
The unsteady Bernoulli equation~\ref{eqnBernoulliUnsteady} gives us:
\begin{equation}
    \rho \frac{\partial \phi}{\partial t} + \frac12 \rho |\nabla \phi|^2 + p_0 + \rho g z = f(t)
    \label{eqnFreeSurfaceDBC}
\end{equation}
on $z = \xi$.
\subsection{Linear Water Waves}
We will assume small amplitude $\zeta \ll h$, and so
\begin{equation*}
    \left|\frac{\partial \zeta}{\partial x}\right|, \left|\frac{\partial \zeta}{\partial y}\right| \ll 1
\end{equation*}
We would like to linearise these equations. We will do this by perturbing about the rest state $\zeta = 0$. Therefore, we ignore any nonlinear terms and we will take Taylor expansions around $z = 0$ for quantities evaluated on the free surface $z = \zeta$.

The resulting linear equations are:
\begin{align}
    \nabla^2 \phi &= 0 & z &\in (-h, 0) \label{eqnLaplaceLinear} \\
    \frac{\partial \phi}{\partial z} &= 0 & z &= -h \label{eqnBottomKBCLinear} \\
    \frac{\partial \zeta}{\partial t} &= \frac{\partial \phi}{\partial z} & z &= 0 \label{eqnFreeSurfaceKBCLinear} \\
    \rho \frac{\partial \phi}{\partial t} + \rho g \zeta &= \tilde{F}(t) & z &= 0 \label{eqnFreeSurfaceDBCLinear}
\end{align}
We will look for a separable solution of the form $\phi(\vec{x}, t) = A(z)B(x) C(y) T(t)$. By taking a Fourier Transform and ignoring $y$ (by restricting to 2D), we end up with solutions of the form:
\begin{equation*}
    \begin{cases}
        \zeta = \zeta_0 \exp(ikx - i\omega t) & \zeta_0 \text{ unknown} \\
        \phi = \Phi(z) \exp(ikx - i\omega t) & \Phi(z) \text{ unknown} \\
    \end{cases}
\end{equation*}
where we assume that real parts are taken for $\zeta$ and $\phi$. The general solution, therefore, is a superposition of an infinite number of Fourier modes (which correspond to travelling waves).

We use equations~\ref{eqnLaplaceLinear} and \ref{eqnBottomKBCLinear}:
\begin{equation*}
    \nabla^2 \phi = 0 \implies \Phi'' - k^2 \Phi = 0
\end{equation*}
so $\Phi(z) = A\cosh [k(z+h)] + B\sinh [k(z+h)]$. Now we must have $\Phi(-h) = 0$ so we have $B = 0$. This is why we chose $z + h$ in the argument. $\Phi(z) = A\cosh[k(z+h)]$.

\Eqnref{eqnFreeSurfaceDBCLinear} implies that $\tilde{F}(t)$ does not depend on $x$ and so its Fourier transform is zero. Equations~\ref{eqnFreeSurfaceKBCLinear} and \ref{eqnFreeSurfaceDBCLinear} imply:
\begin{align*}
    i\omega \xi_0 &+ A k \sinh(kh) = 0 \\
    \rho g \xi_0 &- i\omega A \rho \cosh(kh) = 0
\end{align*}
which is,
\begin{equation*}
    \begin{pmatrix}
        -i\omega & -k\sinh(kh) \\
        \rho g & -i\omega \rho \cosh(kh)
    \end{pmatrix}
    \begin{pmatrix}\xi_0 \\ A\end{pmatrix} = \vec0
\end{equation*}
and so this has a non-zero solution if and only if:
\begin{equation*}
    -\omega^2 \cosh(kh) + gk\sinh(kg) = 0
\end{equation*}
This is the \underline{dispersion relation}:
\begin{equation}
    \omega^2 = gk\tanh(kh)
    \label{eqnDispersionRelation}
\end{equation}
We define the phase speed $c = \frac{\omega}{k}$, and we understand this as the speed of the wave crest. Using this,
\begin{equation*}
    c^2 = \frac{gh \tanh(kh)}{kh}
\end{equation*}
Importantly, water waves are dispersive: waves of different wavelengths travel at different speeds.
\begin{examples}{
        We will consider the following approximations for different scenarios
    }
    \item Deep water waves.
        Here $kh \gg 1$ and this gives $\omega^2 = gk$, $c = \sqrt{\frac{g}{k}}$. Swell waves in the Atlantic Ocean give a period approximately $15s$ so $\omega \approx 0.4$. Here $k^{-1} \approx 60m$ which is much smaller than the ocean depth of $4~km$. Then the wave speed is:
        \begin{equation*}
            c = \frac{\omega}{k} \approx 2000~km~day^{-1}
        \end{equation*}
    \item Shallow-water waves.
        Here $kh \ll 1$, and so $\omega = \sqrt{gh}k$ and $c = \sqrt{gh}$ which is independent of $k$. Therefore, for shallow-water waves, the waves do not disperse.

        In the case of a tsunami, $h \approx 4~km$ but $k^{-1} \approx 500~km$. Then $c \approx 200~ms^{-1}$, which is the speed that commercial planes fly at.
\end{examples}
We now consider the velocity, defined by $\nabla \phi$ where we found:
\begin{equation*}
    \phi = \Re\left[-\frac{i\omega \xi_0}{k} \frac{\cosh[k(z+h)]}{\sin(kh)} e^{i(kx - \omega t)}\right]
\end{equation*}
In the deep water limit, we find that the fluid particles travel in circles. In the shallow water limit, we find large longitudinal motion (along the $x$-axis) and small vertical motion (along $z$).
\subsection{Standing Waves}
Consider water in a rectangular container:
\begin{equation*}
    0 \leq x \leq a;\quad 0\leq y \leq b;\quad z\leq 0
\end{equation*}
Now we require that $\vec{u} \cdot \vec{n} = 0$ on the walls. The equations to solve are now:
\begin{equation*}
    \begin{cases}
        \nabla^2 \phi = 0 & z < 0 \\
        \frac{\partial \phi}{\partial n} = 0 & \text{side walls} \\
        \nabla \phi \to \vec0 & z \to -\infty \\
        \frac{\partial \xi}{\partial t} = \frac{\partial \phi}{\partial z} & z = 0 \\
        \rho \frac{\partial \phi}{\partial t} + \rho g \xi = f(t) & z = 0
    \end{cases}
\end{equation*}
The separation of variables method still works, and we find:
\begin{equation*}
    \phi(x, y, z, t) = A \cos\left(\frac{m \pi x}{a}\right) \cos\left(\frac{n \pi y}{b}\right)e^{kx - i\omega t}
\end{equation*}
now here $m$ and $n$ are integers, so we have discrete wavelengths that solve the equations.
We now find that $k$ is quantised:
\begin{equation*}
    k_{mn}^2 = \frac{m^2 \pi^2}{a^2} + \frac{n^2 \pi^2}{b^2}
\end{equation*}
with $\omega^2 = gk$ as before. These modes are called standing waves (or normal modes).
\subsection{Rayleigh-Taylor Instability}
Consider a container in an unstable state: a layer of water on top of a layer of air. One way to think about this is to instantaneously flip the sign of $g$.

Therefore $\omega^2 = -gk \implies \omega = \pm i\sqrt{gk}$.

The normal modes that were previously given by $e^{i\omega t}$ are now given by $e^{\pm \sqrt{gk}t}$, one of which grows exponentially (resulting in unstable flow).
\section{Rotating Fluid Dynamics}
In this section, we will consider macroscopic flows that are influenced by the rotation of the Earth. This rotation has:
\begin{equation*}
    \omega = \frac{2\pi}{24 \times 60^2} \approx 10^{-4} s^{-1}
\end{equation*}
This rotation has huge effect on the motion of the ocean and atmosphere because the ocean and atmosphere are thin relative to the size of the Earth.
\subsection{Euler Equations in a Rotating Frame}
We will assume $\rho$ constant and $\nabla \cdot \vec{u} = 0$. From the course IA Dynamics and Relativity, the acceleration of a fluid particle in a rotating frame is:
\begin{equation*}
    \frac{D \vec{u}}{D t} + \underbrace{2 \vec{\Omega} \times \vec{u}}_{\text{Coriolis}} + \underbrace{\vec{\Omega} \times \left(\vec{\Omega} \times \vec{x}\right)}_{\text{Centrifugal}}
\end{equation*}
Note that the Centrifugal acceleration is conservative:
\begin{equation*}
    \rho \vec{\Omega} \times \left(\vec{\Omega} \times \vec{x}\right) = \nabla \left[\frac12 \rho |\vec{\Omega} \times \vec{x}|^2\right]
\end{equation*}
and so we can absorb it into $\nabla p$. Further,
\begin{equation*}
    \frac{\vec{\Omega} \times \left(\vec{\Omega} \times \vec{x}\right)}{g} \approx \frac{1}{300}
\end{equation*}
which is small enough that we sometimes neglect it. In this case, the Euler equations are:
\begin{equation}
    \rho \frac{D \vec{u}}{D t} + 2\rho \vec{\Omega} \times \vec{u} = -\nabla p + \rho \vec{g}
    \label{eqnEulerMomRotating}
\end{equation}
Then taking the curl of \eqnref{eqnEulerMomRotating}:
\begin{equation*}
    \frac{D \vec{\omega}}{D t} = ((\vec{\omega} + 2 \Omega) \cdot \nabla) \vec{u}
\end{equation*}
We call $\vec{\omega}$ the relative vorticity, and $2\vec{\Omega}$ the planetary vorticity.
\subsection{Rossby Number}
We want to know how important Coriolis effects are. To do this, consider a flow with characteristic velocity $U$ and length $L$.
\begin{align*}
    \rho \frac{D \vec{u}}{D t} &\sim \rho \frac{U}{L / U} \sim \rho\frac{U^2}{L} \\
    2\rho \vec{\Omega} \times \vec{u} &\sim \rho \Omega U
\end{align*}
and then dividing the two:
\begin{equation*}
    \frac{\rho U^2 / L}{\rho \Omega U} \sim \frac{U}{\Omega L} = Ro
\end{equation*}
This is the dimensionless Rossby number. It quantifies the relative magnitude of inertia and rotation.
\begin{example}
    Consider $\Omega \sim 10^{-4} s^{-1}$, $L \sim 10^6 m$ and $U \sim 10 ms^{-1}$. Then $Ro \sim 10^{-1}$.
\end{example}
When $Ro \ll 1$ then vorticity is dominated by the planetary component.

We consider a length scale $L$ such that:
\begin{equation*}
    H \ll L \ll R
\end{equation*}
and we use local Cartesian coordinates: $\vec{e_x}$ East, $\vec{e_y}$ North and $\vec{e_z}$ locally vertical (out of the sphere), as in figure~\ref{figEarthFluid}.

Let $\overline{\theta}$ be the latitude angle as measured from the Equator.
\begin{align*}
    \vec{\Omega} &= \Omega \left(0, \cos(\overline{\theta}), \sin(\overline{\theta})\right)^T \\
    \vec{u} &= (u,v, w)^T \\
    \therefore 2 \vec{\Omega} \times \vec{u} &= 2\Omega \left(\cos(\overline{\theta})w - \sin(\overline{\theta})v, \sin(\overline{\theta})u, -\cos(\overline{\theta}) u\right)
\end{align*}
and now we make 3 assumptions:
\begin{enumerate}
    \item The order of magnitude of $\vec{u}$ is much larger than the order of magnitude of $\Omega u / g$, and so we neglect the $z$ term of the Coriolis force.
    \item We use that $H \ll L$:
        \begin{align*}
            \nabla \cdot \vec{u} &= 0 \\
            \implies \frac{v}{L} &\sim \frac{w}{H} \\
            \implies w &\sim \frac{H}{l} v \ll v
        \end{align*}
        and so we can discard the term $\cos(\overline{\theta})w$.
    \item $w$ is small, so we can reject the term $\rho \frac{D w}{D t}$ relative to gravity.
\end{enumerate}
Then, by considering the Coriolis parameter $f = 2\Omega \sin(\overline{\theta})$, we have the simplified equations:
\begin{align}
    \frac{D u}{D t} - fv &= -\frac1\rho \frac{\partial p}{\partial x} \label{eqnRotX} \\
    \frac{D v}{D t} + fu &= -\frac1\rho \frac{\partial p}{\partial y} \label{eqnRotY} \\
    0 &= -\frac1\rho\frac{\partial p}{\partial z} - g \label{eqnRotZ} \\
    \frac{\partial u}{\partial x} + \frac{\partial v}{\partial y} + \frac{\partial w}{\partial z} &= 0 \label{eqnRotNoDiv}
\end{align}
We can take a derivative of \eqnref{eqnRotX} with respect to $y$ and a derivative of \eqnref{eqnRotY} with respect to $x$, and subtract them:
\begin{equation*}
    \frac{D \xi}{D t} = (\xi + f) \frac{\partial w}{\partial t}
\end{equation*}
where $\xi$ is the $z$ component of $\vec{\omega}$.
\begin{example}[Flow in a shallow layer]
    Consider flow in a shallow layer of thickness $h(x, y, t)$. Let the bottom of the water be a flat, fixed surface and the upper boundary be air.

    We then find that $p = C(x, y, t) - \rho g z$, and by the condition that $p = p_{atm}$ at $z = h$, $p = \rho g(h - z) + p_0$. Then:
    \begin{align*}
        \frac{\partial p}{\partial x} &= \rho g \frac{\partial h}{\partial x} \\
        \frac{\partial p}{\partial y} &= \rho g \frac{\partial h}{\partial y}
    \end{align*}
    and we write this as $\nabla_H p = \rho g \nabla_H h$ (where $\nabla_H$ is the horizontal gradient, a 2-dimensional vector with derivatives only in $x$ and $y$).

    We note that if $u$ and $v$ are initially independent of $z$, then they are independent of $z$ at all time (because the key equation is independent of $z$).

    Mass conservation takes the form:
    \begin{equation*}
        \frac{d}{dt} \int_A \rho h dA + \oint_{\nabla A} \rho h \vec{u_H} \cdot \vec{n}~dl = 0
    \end{equation*}
    where $\vec{u_h}$ is the vector $\vec{u}$ with $z$-component zero. By the divergence theorem:
    \begin{equation*}
        \frac{D h}{D t} + h \nabla_H \cdot \vec{u_H} = 0
    \end{equation*}
\end{example}
\subsection{Geostrophic Balance}
Consider a steady flow with $Ro \ll 1$. This means we can ignore the material derivative terms in equations~\ref{eqnRotX} and \ref{eqnRotY}, resulting in:
\begin{align*}
    -f_v &= -\frac1\rho \frac{\partial p}{\partial x} \\
    f_u &= -\frac1\rho \frac{\partial p}{\partial y} \\
\end{align*}
we recognise this as the form of a streamfunction, re-writing as:
\begin{align*}
    u &= \frac{\partial}{\partial y} \left[-\frac{gh}{f}\right] \\
    v &= -\frac{\partial}{\partial x} \left[-\frac{gh}{f}\right]
\end{align*}
the streamfunction becomes:
\begin{equation*}
    \psi = -\frac{gh}{f} = -\frac{p}{\rho f}
\end{equation*}
and so the flow is along isobars (lines of constant pressure).
\subsection{Potential Vorticity}
Recall that:
\begin{equation*}
    \frac{D}{D t} \left(\xi + f\right) = \frac{D \xi}{D t} = -(\xi + f) \nabla_H \vec{u_H}
\end{equation*}
and we have the similar equation:
\begin{equation*}
    \frac{D h}{D t} = -h \nabla_H \cdot \vec{u_H}
\end{equation*}
and so we find a conserved quantity:
\begin{equation*}
    \frac{D}{D t}\left(\frac{\xi + f}{h}\right) = 0
\end{equation*}
where this quantity is the \underline{potential vorticity}.
\subsection{Linearised Evolution}
In this subsection we take $Ro \ll 1$ but allow unsteady flow. Considering a small perturbation, $h = h_0 + \eta(x, y, t)$ where $|\eta| \ll h$, we transform equations~\ref{eqnRotX} and \ref{eqnRotY} to:
\begin{align}
    \frac{\partial u}{\partial t} - fv &= -\frac1\rho \frac{\partial p}{\partial x} = -g \frac{\partial h}{\partial x} \label{eqnLinRotX} \\
    \frac{\partial v}{\partial t} + fu &= -\frac1\rho \frac{\partial p}{\partial y} = -g \frac{\partial h}{\partial y} \label{eqnLinRotY} \\
    \intertext{and the mass conservation equation becomes:}
    \frac{\partial \eta}{\partial t} &+ h_0 \left(\frac{\partial u}{\partial x} + \frac{\partial v}{\partial y}\right) = 0 \label{eqnLinRotMassConserve}
\end{align}
By linearising the potential vorticity, we find:
\begin{equation*}
    \xi - \eta \frac{f}{h_0} = \text{const}
\end{equation*}
Then we can find an equation in $\eta$ by considering $h_0 \frac{\partial}{\partial x}$ of \eqnref{eqnLinRotX} plus $h_0 \frac{\partial}{\partial y}$ of \eqnref{eqnLinRotY} minus $\frac{\partial}{\partial t}$ of \eqnref{eqnLinRotMassConserve}:
\begin{equation}
    \frac{\partial^{2}\eta}{\partial t^{2}} - c^2 \nabla^2 \eta + f^2 \eta = f^2 \eta_0 - h_0 f \xi_0
    \label{eqnLinRot}
\end{equation}
this invites us to consider a characteristic length scale,
\begin{equation*}
    a = \frac{c}{f} = \frac{\sqrt{g h_0}}{f}
\end{equation*}
which is the \underline{Rossby Radius of Deformation}.
